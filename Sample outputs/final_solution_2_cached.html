[ 2 ] Select mode of operation and define input¶

  • There are two modes of operation, DEFAULT and CACHED.
  • CACHED mode is faster as all the images are already encoded and similarities are precalculated.
  • For 'cached' mode, input file should be an integer value (as encoding for this is already saved).
  • For 'default' mode, input file should be the complete path to the image file.
  • 'sim_count' specifies the number of similar images that are required.

""" CACHED mode execution """

if mode == 'cached':
    
    # load the sorted cache
    
    if sorted_cache_not_loaded:
        
        with open(sorted_cache) as file:
            cached_sim_mat_sorted = json.load(file)
        
        sorted_cache_not_loaded = False
    
    # find N similar images from the cache
    n_similar = cached_sim_mat_sorted[input_file][1: sim_count+1]

else:
    
    """ DEFAULT mode execution """
    
    # load the encoder if not already loaded
    if encoder_not_loaded:
        encoder = tf.keras.models.load_model(encoder_filename)
        encoder_not_loaded = False
    
    # preprocess the input image to get the correct dimensions and then normalize it
    image = tf.keras.preprocessing.image.load_img(input_file, target_size=(256, 256), interpolation='bilinear')
    image = tf.keras.preprocessing.image.img_to_array(image)
    image /= 255
    
    # get the image encoding
    encoding = encoder.predict(image[None, :, :, :]).squeeze().reshape(4096,)
    
    # calculate cosine similarity of this image encoding with all the cached encodings
    cached_image_encodings = np.load(cache_filename)
    cos_sim_with_others = []
    for i in range(cached_image_encodings.shape[0]):
        cur_cos_sim = tf.keras.losses.cosine_similarity(encoding, cached_image_encodings[i]).numpy()
        cos_sim_with_others.append([str(i), float(cur_cos_sim)])
    
    # sort the images in decreasing order of similarity
    cos_sim_with_others.sort(key=lambda x: x[1])
    
    # select N similar images from it
    n_similar = cos_sim_with_others[: sim_count]

""" Plot input image and N similar image(s) """

# get filename of input image
if mode == 'cached':
    filepath = image_dir + str(input_file) + '.jpg'
else:
    filepath = input_file

# get filename of similar images
n_similar_images = [image_dir + file + '.jpg' for file, similarity in n_similar]

# get the required number of rows and columns for creating subplots
rows = ceil(len(n_similar_images)/5)
cols = 5

# turn off x and y ticks
plt.xticks([])
plt.yticks([])

# show the input image
plt.imshow(np.asarray(Image.open(filepath)))

# show the similar images
fig, ax = plt.subplots(rows, cols, figsize=(10, rows*2.5), gridspec_kw = {'hspace': 0.2, 'wspace': 0.1})
i = 0
for x in range(rows):
    for y in range(cols):
        ax[x, y].set_xticks([])
        ax[x, y].set_yticks([])
        if i >= len(n_similar_images):
            ax[x, y].axis('off')
        else:
            ax[x, y].imshow(np.asarray(Image.open(n_similar_images[i])))
        i += 1
plt.show()
